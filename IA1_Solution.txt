create table PROTYP(
PID integer,
name varchar(30),
constraint PID_PK_CONSTRAINT primary key(PID),
check(name in ('residential','Industrial property', 'Commercial property')),
constraint PID_FK_ERR
foreign key(PID) references PR(PID)
)engine=InnoDB;
